int s1index_i=0; // can be commented out as it is initialised in the for loop

s1index_i=s1cdncol;

    for(s1index_i=s1cdncol;s1index_i>1;s1index_i--)

    array_dsply[0]=array_src[s1index_j]; // introduce the new element to the display array at its head/first position
